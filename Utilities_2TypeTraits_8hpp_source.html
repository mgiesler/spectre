571 template <typename TT, typename... TArgs>
572 class is_callable {
573  // The reason we have private before public here is that we have static member
574  // functions and since this is meant to be a super lightweight helper class
575  // it's better to break convention than increase code size.
576  private:
578  // We pass an int here to disambiguate the two possible templates and have the
579  // compiler prefer the first one. If it cannot be used because there's no
580  // call operator, then it uses the second one.
581  template <typename T, typename... Args>
582  static auto test_callable(int)
583  -> decltype(std::declval<T>()(std::declval<Args>()...), std::true_type());
584 
585  template <typename, typename...>
586  static auto test_callable(...) -> std::false_type;
588 
589  public:
591  static constexpr bool value = decltype(test_callable<TT, TArgs...>(0))::value;
593  using type = std::integral_constant<bool, value>;
594 };
