69 inline SegmentId SegmentId::id_of_parent() const {
70  ASSERT(0 != refinement_level_,
71  "Cannot call id_of_parent() on root refinement level!");
72  // right-shifting index_ by 1 gives us floor(index_/2).
73  // The parent has half as many segments as the child.
74  return SegmentId(refinement_level_ - 1, index_ >> 1);
75 }
76 
77 inline SegmentId SegmentId::id_of_child(Side side) const {
78  // We cannot ASSERT on the maximum level because it's only known at runtime
79  // and only known elsewhere in the code, not by SegmentId. I.e. SegmentId is
80  // too low-level to know about this.
81  // The child has twice as many segments as the parent, so for a particular
82  // parent segment, there is both an upper and lower child segment.
83  if (Side::Lower == side) {
84  // left-shifting index_ by 1 gives us index*2.
85  return SegmentId(refinement_level_ + 1, index_ << 1);
86  }
87  // left-shifting index_ by 1, followed by OR with 1, gives us index*2 +1.
88  return SegmentId(refinement_level_ + 1, 1 | (index_ << 1));
89 }

96 inline double SegmentId::endpoint(Side side) const {
97  if (Side::Lower == side) {
98  return -1.0 + (2.0 * index_) / two_to_the(refinement_level_);
99  }
100  return -1.0 + (2.0 * index_ + 2.0) / two_to_the(refinement_level_);
101 }
