54 TimeDelta RungeKutta3::update_u(
55  Vars& u,
56  const std::deque<std::tuple<Time, Vars, DerivVars>>& history,
57  const TimeDelta& time_step) const noexcept {
58  const size_t substep = history.size() - 1;
59  const auto& vars = std::get<1>(history.back());
60  const auto& dt_vars = std::get<2>(history.back());
61  const auto& U0 = std::get<1>(history[0]);
62 
63  switch (substep) {
64  case 0: {
65  // from (5.32) of Hesthaven
66  // v^(1) = u^n + dt*RHS(u^n,t^n)
67  // On entry V = u^n, U0 = u^n, rhs0 = RHS(u^n,t^n),
68  // time = t^n
69  u += time_step.value() * dt_vars;
70  return time_step;
71  // On exit v = v^(1), time = t^n + dt
72  }
73  case 1: {
74  // from (5.32) of Hesthaven
75  // v^(2) = (1/4)*( 3*u^n + v^(1) + dt*RHS(v^(1),t^n + dt) )
76  // On entry V = v^(1), U0 = u^n, rhs0 = RHS(v^(1),t^n + dt),
77  // time = t^n + dt
78  u += 0.25 * (3.0 * (U0 - vars) + time_step.value() * dt_vars);
79  return -time_step / 2;
80  // On exit v = v^(2), time = t^n + (1/2)*dt
81  }
82  case 2: {
83  // from (5.32) of Hesthaven
84  // u^(n+1) = (1/3)*( u^n + 2*v^(2) + 2*dt*RHS(v^(2),t^n + (1/2)*dt) )
85  // On entry V = v^(2), U0 = u^n, rhs0 = RHS(v^(2),t^n + (1/2)*dt),
86  // time = t^n + (1/2)*dt
87  u += (1.0 / 3.0) * (U0 - vars + 2.0 * time_step.value() * dt_vars);
88  return time_step / 2;
89  // On exit v = u^(n+1), time = t^n + dt
90  }
91  default:
92  ERROR("Bad substep value in RK3: " << substep);
93  }
94 }
