16 template <typename Stepper, typename F>
17 void take_step(
18  Time& time,
19  double& y,
20  std::deque<std::tuple<Time, double, double>>& history,
21  const Stepper& stepper,
22  F&& rhs,
23  const TimeDelta& step_size) noexcept {
24  TimeDelta accumulated_step_dt = step_size * 0;
25  for (size_t substep = 0; substep < stepper.number_of_substeps(); ++substep) {
26  history.emplace_back(time, y, rhs(y));
27  const TimeDelta substep_dt = stepper.update_u(y, history, step_size);
28  accumulated_step_dt += substep_dt;
29  time += substep_dt;
30  }
31  CHECK(accumulated_step_dt == step_size);
32  while (history.size() > stepper.number_of_past_steps()) {
33  history.pop_front();
34  }
35 }

37 template <typename Stepper, typename F1, typename F2>
38 void initialize_history(
39  Time time,
40  std::deque<std::tuple<Time, double, double>>& history,
41  const Stepper& stepper,
42  F1&& analytic,
43  F2&& rhs,
44  TimeDelta step_size) noexcept {
45  for (size_t j = 0; j < stepper.number_of_past_steps(); ++j) {
46  ASSERT(time.slab() == step_size.slab(), "Slab mismatch");
47  if ((step_size.is_positive() and time.is_at_slab_start()) or
48  (not step_size.is_positive() and time.is_at_slab_end())) {
49  const Slab new_slab = time.slab().advance_towards(-step_size);
50  time = time.with_slab(new_slab);
51  step_size = step_size.with_slab(new_slab);
52  }
53  time -= step_size;
54  history.emplace_front(time, analytic(time.value()),
55  rhs(analytic(time.value())));
56  }
57 }

70 template <typename Stepper>
71 void integrate_test(const Stepper& stepper, const double integration_time,
72  const double epsilon) noexcept {
73  auto analytic = [](double t) { return sin(t); };
74  auto rhs = [](double v) { return sqrt(1. - square(v)); };
75 
76  Approx approx = Approx::custom().epsilon(epsilon);
77 
78  const size_t num_steps = 800;
79  const Slab slab = integration_time > 0
80  ? Slab::with_duration_from_start(0., integration_time)
81  : Slab::with_duration_to_end(0., -integration_time);
82  const TimeDelta step_size = integration_time > 0
83  ? slab.duration() / num_steps
84  : -slab.duration() / num_steps;
85 
86  Time time = integration_time > 0 ? slab.start() : slab.end();
87  double y = analytic(time.value());
88  std::deque<std::tuple<Time, double, double>> history;
89 
90  if (not stepper.is_self_starting()) {
91  initialize_history(time, history, stepper, analytic, rhs, step_size);
92  }
93 
94  for (size_t i = 0; i < num_steps; ++i) {
95  take_step(time, y, history, stepper, rhs, step_size);
96  CHECK(y == approx(analytic(time.value())));
97  }
98 }

129 template <typename Stepper>
130 void stability_test(const Stepper& stepper) noexcept {
131  const size_t num_steps = 5000;
132  const double bracket_size = 1.1;
133 
134  // This is integrating dy/dt = -2y, which is chosen so that the stable
135  // step size for Euler's method is 1.
136 
137  // Stable region
138  {
139  const Slab slab = Slab::with_duration_from_start(
140  0., num_steps * stepper.stable_step() / bracket_size);
141  const TimeDelta step_size = slab.duration() / num_steps;
142 
143  Time time = slab.start();
144  double y = 1.;
145  std::deque<std::tuple<Time, double, double>> history;
146  if (not stepper.is_self_starting()) {
147  initialize_history(time, history, stepper,
148  [](double t) { return exp(-2. * t); },
149  [](double v) { return -2. * v; },
150  step_size);
151  }
152 
153  for (size_t i = 0; i < num_steps; ++i) {
154  take_step(time, y, history, stepper, [](double v) { return -2. * v; },
155  step_size);
156  CHECK(std::abs(y) < 10.);
157  }
158  }
159 
160  // Unstable region
161  {
162  const Slab slab = Slab::with_duration_from_start(
163  0., num_steps * stepper.stable_step() * bracket_size);
164  const TimeDelta step_size = slab.duration() / num_steps;
165 
166  Time time = slab.start();
167  double y = 1.;
168  std::deque<std::tuple<Time, double, double>> history;
169  if (not stepper.is_self_starting()) {
