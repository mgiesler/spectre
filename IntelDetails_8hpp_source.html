15 template <typename T, typename S, size_t Size>
16 void increment_tensor_index(std::array<T, Size>& tensor_index,
17  const std::array<S, Size>& dims) {
18  for (size_t i = 0; i < Size; ++i) {
19  if (++tensor_index[i] < static_cast<T>(dims[i])) {
20  break;
21  }
22  tensor_index[i] = 0;

56 template <typename Symm, typename IndexLs, typename T>
57 T canonicalize_tensor_index(T arr) {
58  static constexpr auto symm = ::make_array_from_list<Symm>();
59  static constexpr size_t rank = arr.size();
60  for (size_t i = 0; i < rank; ++i) {
61  std::swap(arr[i], arr[index_to_swap(arr, symm, i, i)]);
62  }
63  return arr;
64 }

66 template <
67  typename Symm, typename IndexLs, typename NumComps,
68  typename std::enable_if_t<(tmpl::size<IndexLs>::value > 0)>* = nullptr>
69 std::array<int, NumComps::value> compute_collapsed_to_storage() {
70  static constexpr auto dims = ::make_array_from_list<IndexLs>();
71  static constexpr auto rank = tmpl::size<IndexLs>::value;
72  std::array<int, NumComps::value> collapsed_to_storage;
73  auto tensor_index = make_array<rank>(0);
74  int count{0};
75  for (auto& current_storage_index : collapsed_to_storage) {
76  // Compute canonical tensor_index, which, for symmetric get_tensor_index is
77  // in decreasing numerical order, e.g. (3,2) rather than (2,3).
78  auto canonical_tensor_index =
79  canonicalize_tensor_index<Symm, IndexLs>(tensor_index);
80  // If the tensor_index was already in the canonical form, then it must be a
81  // new unique entry and we add it to collapsed_to_storage_ as a new
82  // integer, thus increasing the size_. Else, the StorageIndex has already
83  // been determined so we look it up in the existing collapsed_to_storage
84  // table.
85  if (tensor_index == canonical_tensor_index) {
86  current_storage_index = count;
87  ++count;
88  } else {
89  current_storage_index = collapsed_to_storage[::compute_collapsed_index(
90  canonical_tensor_index, dims)];
91  }
92  // Move to the next tensor_index.
93  increment_tensor_index(tensor_index, dims);
94  }
95  return collapsed_to_storage;
96 }
