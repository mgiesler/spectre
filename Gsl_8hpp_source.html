92 template <class T, class U>
93 SPECTRE_ALWAYS_INLINE T narrow(U u) {
94  T t = narrow_cast<T>(u);
95  if (static_cast<U>(t) != u) {
96  ERROR("Failed to cast " << u << " of type " << pretty_type::get_name<U>()
97  << " to type " << pretty_type::get_name<T>());
98  }
99  if (not details::is_same_signedness<T, U>::value and
100  ((t < T{}) != (u < U{}))) {
101  ERROR("Failed to cast " << u << " of type " << pretty_type::get_name<U>()
102  << " to type " << pretty_type::get_name<T>());
103  }
104  return t;
105 }
