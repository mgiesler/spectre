137 template <typename T>
138 constexpr auto CoordinateMap<SourceFrame, TargetFrame, Maps...>::inv_jacobian(
139  const tnsr::I<T, dim, SourceFrame>& source_point) const
140  -> Tensor<T, tmpl::integral_list<std::int32_t, 2, 1>,
141  index_list<SpatialIndex<dim, UpLo::Up, SourceFrame>,
142  SpatialIndex<dim, UpLo::Lo, TargetFrame>>> {
143  std::array<T, dim> mapped_point = make_array<T, dim>(source_point);
144 
145  Tensor<T, tmpl::integral_list<std::int32_t, 2, 1>,
146  index_list<SpatialIndex<dim, UpLo::Up, SourceFrame>,
147  SpatialIndex<dim, UpLo::Lo, TargetFrame>>>
148  inv_jac{};
149 
150  tuple_transform(
151  maps_,
152  [&inv_jac, &mapped_point](const auto& map, auto index,
153  const std::tuple<Maps...>& maps) {
154  constexpr const size_t count = decltype(index)::value;
155  auto temp_inv_jac = map.inv_jacobian(mapped_point);
156 
157  if (LIKELY(count != 0)) {
158  mapped_point =
159  std::get<(count != 0 ? count - 1 : 0)>(maps)(mapped_point);
160  std::array<T, dim> temp{};
161  for (size_t source = 0; source < dim; ++source) {
162  for (size_t target = 0; target < dim; ++target) {
163  gsl::at(temp, target) =
164  inv_jac.get(source, 0) * temp_inv_jac.get(0, target);
165  for (size_t dummy = 1; dummy < dim; ++dummy) {
166  gsl::at(temp, target) += inv_jac.get(source, dummy) *
167  temp_inv_jac.get(dummy, target);
168  }
169  }
170  for (size_t target = 0; target < dim; ++target) {
171  inv_jac.get(source, target) = std::move(gsl::at(temp, target));
172  }
173  }
174  } else {
175  for (size_t source = 0; source < dim; ++source) {
176  for (size_t target = 0; target < dim; ++target) {
177  inv_jac.get(source, target) =
178  std::move(temp_inv_jac.get(source, target));
179  }
180  }
181  }
182  },
183  maps_);
184  return inv_jac;
185 }
