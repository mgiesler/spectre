29 // applies the function f(out, it) to each item from begin to end, separated
30 // by commas and surrounded by parens
31 template <typename ForwardIt, typename Func>
32 inline void print_helper(std::ostream& out, ForwardIt&& begin, ForwardIt&& end,
33  Func f) {
34  out << "(";
35  if (begin != end) {
36  while (true) {
37  f(out, begin++);
38  if (begin == end) {
39  break;
40  }
41  out << ",";
42  }
43  }
44  out << ")";
45 }
