379 template <typename VT, bool VF>
380 DataVector& DataVector::operator=(const blaze::Vector<VT, VF>& expression) {
381  if (owning_ and (~expression).size() != size()) {
382  size_ = (~expression).size();
383  owned_data_ = InternalStorage_t(size_);
384  data_ = decltype(data_){owned_data_.data(), size_};
385  } else if (not owning_) {
386  ASSERT((~expression).size() == size(), "Must copy into same size");
387  }
388  data_ = expression;
389  return *this;
390 }
