572 template <typename TT, typename... TArgs>
573 class is_callable {
574  // The reason we have private before public here is that we have static member
575  // functions and since this is meant to be a super lightweight helper class
576  // it's better to break convention than increase code size.
577  private:
579  // We pass an int here to disambiguate the two possible templates and have the
580  // compiler prefer the first one. If it cannot be used because there's no
581  // call operator, then it uses the second one.
582  template <typename T, typename... Args>
583  static auto test_callable(int) noexcept
584  -> decltype(std::declval<T>()(std::declval<Args>()...), std::true_type());
585 
586  template <typename, typename...>
587  static auto test_callable(...) noexcept -> std::false_type;
589 
590  public:
592  static constexpr bool value = decltype(test_callable<TT, TArgs...>(0))::value;
594  using type = std::integral_constant<bool, value>;
595 };
