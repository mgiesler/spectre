347  CASSERT(size_ > (variable_offset + TagToAdd::type::size() - 1) *

355  for (size_t i = 0; i < TagToAdd::type::size(); ++i) {
356  var[i].set_data_ref(
357  &variable_data_[(variable_offset + i) * number_of_grid_points_],
358  number_of_grid_points_);
359  }
360  add_reference_variable_data(typelist<Rest...>{},
361  variable_offset + TagToAdd::type::size());
362 }
