422 template <template <typename...> class TagsLs, typename... Tags>
423 template <typename RemoveTags, typename AddTags, typename AddComputeItems,
424  typename Box, typename... Args>
425 constexpr auto DataBox<TagsLs<Tags...>>::create_from(const Box& box,
426  Args&&... args) {
427  static_assert(tt::is_a_v<::db::DataBox, Box>,
428  "create_from must receive a DataBox as its first argument");
429  static_assert(tt::is_a_v<tmpl::list, RemoveTags>,
430  "RemoveTags must by a typelist");
431  static_assert(tt::is_a_v<tmpl::list, AddTags>, "AddTags must by a typelist");
432  static_assert(tt::is_a_v<tmpl::list, AddComputeItems>,
433  "AddComputeItems must by a typelist");
434  static_assert(
435  tmpl::all<AddComputeItems, is_compute_item<tmpl::_1>>::value,
436  "Cannot add any ComputeItemTag in the AddTags list, must use the "
437  "AddComputeItems list.");
438  using old_tags_list = typename Box::tags_list;
439 
440  // Build list of compute items in Box::tags_list that are not in RemoveTags
441  using compute_items_to_keep = tmpl::filter<
442  old_tags_list,
443  tmpl::and_<
444  db::is_compute_item<tmpl::_1>,
445  tmpl::not_<tmpl::bind<tmpl::found, tmpl::pin<RemoveTags>,
446  tmpl::bind<std::is_same, tmpl::parent<tmpl::_1>,
447  tmpl::bind<tmpl::pin, tmpl::_1>>>>>>;
448 
449  // Build list of tags where we expand the tags inside Variables<Tags...>
450  // objects. This is needed since we actually want those tags to be part of the
451  // DataBox type as well
452  using full_remove_tags =
453  tmpl::fold<RemoveTags, tmpl::list<>,
454  tmpl::bind<tmpl::append, tmpl::_state,
455  detail::extract_dependent_items<tmpl::_element>>>;
456  using full_items =
457  tmpl::fold<AddTags, tmpl::list<>,
458  tmpl::bind<tmpl::append, tmpl::_state,
459  detail::extract_dependent_items<tmpl::_element>>>;
460  using full_compute_items =
461  tmpl::fold<AddComputeItems, tmpl::list<>,
462  tmpl::bind<tmpl::append, tmpl::_state,
463  detail::extract_dependent_items<tmpl::_element>>>;
464  using remaining_tags =
465  tmpl::fold<full_remove_tags, old_tags_list,
466  tmpl::bind<tmpl::remove, tmpl::_state, tmpl::_element>>;
467  using new_tags = tmpl::append<remaining_tags, full_items, full_compute_items>;
468  using sorted_tags = ::db::get_databox_list<new_tags>;
469  return DataBox<sorted_tags>(box, remaining_tags{}, AddTags{},
470  AddComputeItems{}, compute_items_to_keep{},
471  std::forward<Args>(args)...);
472 }

655  // Reset what depends on me next

657  tmpl::size<typename current_vertex::argument_tags>::value == 0,

663  // Reset next item in current list (level)
664  ResetComputeItems<DependencyGraph, tmpl::pop_front<Vertices>>::apply(data);
