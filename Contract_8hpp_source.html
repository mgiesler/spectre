44 template <int I, typename Index1, typename Index2>
45 struct ComputeContractionImpl {
46  template <typename... LhsIndices, typename T, typename S>
47  static SPECTRE_ALWAYS_INLINE typename T::type apply(S tensor_index,
48  const T& t1) {
49  tensor_index[Index1::value] = I;

52  ComputeContractionImpl<I - 1, Index1, Index2>::template apply<

54  }

145  SPECTRE_ALWAYS_INLINE type
