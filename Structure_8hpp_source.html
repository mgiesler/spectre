46 template <size_t NumIndices>
47 struct ComponentNameImpl {
48  template <typename Structure, typename T>
49  static std::string apply(
50  const std::array<T, NumIndices>& tensor_index,
51  const std::array<std::string, NumIndices>& axis_labels) {
52  const size_t storage_index = Structure::get_storage_index(tensor_index);
53  std::array<std::string, Structure::rank()> labels = axis_labels;
54  constexpr auto index_dim = Structure::dims();
55  for (size_t i = 0; i < Structure::rank(); ++i) {
56  if (labels[i].length() == 0) { // NOLINT
57  if (Structure::index_types()[i] == IndexType::Spacetime) { // NOLINT
58  switch (gsl::at(index_dim, i)) {
59  case 2:
60  labels[i] = "tx"; // NOLINT
61  break;
62  case 3:
63  labels[i] = "txy"; // NOLINT
64  break;
65  case 4:
66  labels[i] = "txyz"; // NOLINT
67  break;
68  default:
69  ERROR("Tensor dim["
70  << i << "] must be 1,2,3, or 4 for default axis_labels. "
71  "Either pass a string or extend the function.");
72  }
73  } else {
74  switch (gsl::at(index_dim, i)) {
75  case 1:
76  labels[i] = "x"; // NOLINT
77  break;
78  case 2:
79  labels[i] = "xy"; // NOLINT
80  break;
81  case 3:
82  labels[i] = "xyz"; // NOLINT
83  break;
84  default:
85  ERROR("Tensor dim["
86  << i << "] must be 1,2, or 3 for default axis_labels. "
87  "Either pass a string or extend the function.");
88  }
89  }
90  } else {
91  if (axis_labels[i].length() != gsl::at(index_dim, i)) { // NOLINT
92  ERROR("Dimension mismatch: Tensor has dim = "
93  << gsl::at(index_dim, i) << ", but you specified "
94  << axis_labels[i].length() << " different labels in "
95  << axis_labels[i]);
96  }
97  }
98  }
99  // Create string labeling get_tensor_index
100  std::stringstream ss;
101  const auto& canonical_tensor_index =
102  Structure::get_canonical_tensor_index(storage_index);
103  for (size_t r = 0; r < Structure::rank(); ++r) {
104  ss << labels[r][canonical_tensor_index[r]]; // NOLINT
105  }
106  return ss.str();
107  }
108 };

120 template <typename T, typename S, std::size_t Rank>
121 inline constexpr std::size_t compute_collapsed_index(
122  const std::array<T, Rank>& tensor_index, const std::array<S, Rank> dims,
123  const size_t i = 0) noexcept {
124  static_assert(tt::is_integer_v<T>,
125  "The tensor index array must hold integer types.");
126  static_assert(tt::is_integer_v<S>, "The dims array must hold integer types.");
127  return i < Rank ? (gsl::at(tensor_index, i) +
128  gsl::at(dims, i) *
129  compute_collapsed_index(tensor_index, dims, i + 1))
130  : 0;
131 }

158 template <typename Symm, typename... Indices>
159 struct Structure {
160  static_assert(
161  TensorMetafunctions::check_index_symmetry<Symm,
162  tmpl::list<Indices...>>::value,
163  "Cannot construct a Tensor with a symmetric pair that are not the same.");
164  static_assert(tmpl::size<Symm>::value == sizeof...(Indices),
165  "The number of indices in Symmetry do not match the number of "
166  "indices given to the Structure.");
167  static_assert(
168  cpp17::conjunction<tt::is_tensor_index_type<Indices>...>::value,
169  "All Indices passed to Structure must be of type TensorIndexType.");
170 
171  using index_list = tmpl::list<Indices...>;
172  using num_of_components =
173  TensorMetafunctions::number_of_components<Symm, index_list>;
174  using NumberOfIndependentComponents =
175  TensorMetafunctions::independent_components<Symm, index_list>;
176 
177 #ifdef __INTEL_COMPILER
178  // The Intel compiler is embarrassingly terrible at compiling metaprograms,
179  // possibly because of a lack of memoization, but without having access to
180  // their compiler source we cannot be sure. Unfortunately ICC is also terrible
181  // at supporting constant expressions and so moving these computations to
182  // constant expressions is also not feasible (it's been attempted). As a
183  // result we do the computations of the arrays at compile time when using GCC
184  // or Clang and at runtime when using Intel. It should be possible to later
185  // make the arrays static so they are only computed at most once during
186  // evolution. However, if Tensor has the structure as a static member variable
187  // then this is irrelevant anyway.
188  //
189  // ICC chokes on the computation of the collapsed_to_storage_ array.
190  // Specifically it cannot handle the IncrementTensorIndex call. We've tested
191  // several other implementations of this metafunction but they all result in
192  // the same time with GCC and Clang but impossibly long compilation with ICC.
193  const std::array<size_t,
194  number_of_components<Symm, tmpl::list<Indices...>>::value>
195  collapsed_to_storage_ = Tensor_detail::compute_collapsed_to_storage<
196  Symm, tmpl::list<Indices...>,
197  number_of_components<Symm, index_list>>();
198  const std::array<
199  std::array<size_t, sizeof...(Indices) == 0 ? 1 : sizeof...(Indices)>,
200  IndependentComponents<Symm, tmpl::list<Indices...>>::value>
201  storage_to_tensor_ = Tensor_detail::compute_storage_to_tensor<
202  Symm, tmpl::list<Indices...>,
203  IndependentComponents<Symm, index_list>>(collapsed_to_storage_);
204  const std::array<size_t,
205  IndependentComponents<Symm, tmpl::list<Indices...>>::value>
206  multiplicity_ = Tensor_detail::compute_multiplicity<
207  IndependentComponents<Symm, tmpl::list<Indices...>>>(
208  collapsed_to_storage_);
209 #else
210  using collapsed_to_storage_list =
211  TensorMetafunctions::compute_collapsed_to_storage<index_list, Symm,
212  num_of_components>;
213  using storage_to_tensor_index_list =
214  TensorMetafunctions::compute_storage_to_tensor<
215  Symm, index_list, collapsed_to_storage_list,
216  NumberOfIndependentComponents>;
217 
218  using multiplicity_list =
219  TensorMetafunctions::compute_multiplicity<collapsed_to_storage_list,
220  NumberOfIndependentComponents>;
221 #endif
