97 template <typename Vars, typename DerivVars>
98 TimeDelta AdamsBashforthN::update_u(
99  Vars& u,
100  const std::deque<std::tuple<Time, Vars, DerivVars>>& history,
101  const TimeDelta& time_step) const noexcept {
102  ASSERT(is_self_starting_ or target_order_ == history.size(),
103  "Length of history should be the order, so "
104  << target_order_ << ", but is: " << history.size());
105  ASSERT(history.size() <= target_order_,
106  "Length of history (" << history.size() << ") "
107  << "should not exceed target order (" << target_order_ << ")");
108 
109  // Cast the overloaded get function to a function pointer taking a
110  // history entry to a time reference. This is needed to prevent the
111  // boost::make_transform_iterator calls from being ambiguous.
112  const Time& (*const get_time)(const decltype(history[0])&) = std::get<0>;
113  const auto& coefficients = get_coefficients(
114  boost::make_transform_iterator(history.cbegin(), get_time),
115  boost::make_transform_iterator(history.cend(), get_time), time_step);
116 
117  switch (history.size()) {
118  case 1: {
119  u += time_step.value() * std::get<2>(history[0]);
120  return time_step;
121  }
122  case 2: {
123  u += time_step.value() * (coefficients[1] * std::get<2>(history[0]) +
124  coefficients[0] * std::get<2>(history[1]));
125  return time_step;
126  }
127  case 3: {
128  u += time_step.value() * (coefficients[2] * std::get<2>(history[0]) +
129  coefficients[1] * std::get<2>(history[1]) +
130  coefficients[0] * std::get<2>(history[2]));
131  return time_step;
132  }
133  case 4: {
134  u += time_step.value() * (coefficients[3] * std::get<2>(history[0]) +
135  coefficients[2] * std::get<2>(history[1]) +
136  coefficients[1] * std::get<2>(history[2]) +
137  coefficients[0] * std::get<2>(history[3]));
138  return time_step;
139  }
140  case 5: {
141  u += time_step.value() * (coefficients[4] * std::get<2>(history[0]) +
142  coefficients[3] * std::get<2>(history[1]) +
143  coefficients[2] * std::get<2>(history[2]) +
144  coefficients[1] * std::get<2>(history[3]) +
145  coefficients[0] * std::get<2>(history[4]));
146  return time_step;
147  }
148  case 6: {
149  u += time_step.value() * (coefficients[5] * std::get<2>(history[0]) +
150  coefficients[4] * std::get<2>(history[1]) +
151  coefficients[3] * std::get<2>(history[2]) +
152  coefficients[2] * std::get<2>(history[3]) +
153  coefficients[1] * std::get<2>(history[4]) +
154  coefficients[0] * std::get<2>(history[5]));
155  return time_step;
156  }
157  case 7: {
158  u += time_step.value() * (coefficients[6] * std::get<2>(history[0]) +
159  coefficients[5] * std::get<2>(history[1]) +
160  coefficients[4] * std::get<2>(history[2]) +
161  coefficients[3] * std::get<2>(history[3]) +
162  coefficients[2] * std::get<2>(history[4]) +
163  coefficients[1] * std::get<2>(history[5]) +
164  coefficients[0] * std::get<2>(history[6]));
165  return time_step;
166  }
167  case 8: {
168  u += time_step.value() * (coefficients[7] * std::get<2>(history[0]) +
169  coefficients[6] * std::get<2>(history[1]) +
170  coefficients[5] * std::get<2>(history[2]) +
171  coefficients[4] * std::get<2>(history[3]) +
172  coefficients[3] * std::get<2>(history[4]) +
173  coefficients[2] * std::get<2>(history[5]) +
174  coefficients[1] * std::get<2>(history[6]) +
175  coefficients[0] * std::get<2>(history[7]));
176  return time_step;

178  default:

183 template <typename Iterator>
184 std::vector<double> AdamsBashforthN::get_coefficients(
185  const Iterator& times_begin, const Iterator& times_end,
186  const TimeDelta& step) noexcept {
187  std::vector<double> steps;
188  // This may be slightly more space than we need, but we can't get
189  // the exact amount without iterating through the iterators, which
190  // is not necessarily cheap depending on the iterator type.
191  steps.reserve(maximum_order);
192  for (auto t = times_begin; std::next(t) != times_end; ++t) {
193  steps.push_back((*std::next(t) - *t) / step);
194  }
195  steps.push_back(1.);
196  return get_coefficients_impl(steps);
197 }
